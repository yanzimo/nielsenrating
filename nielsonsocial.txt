                                              <h1 >Weekly Top Ten</h1>

                <div class="page-content">
                  <p>Social Content Ratings Weekly Top Ten lists for <a href="/socialcontentratings/weekly/#SeriesSpecials">Series and Specials</a> and <a href="/socialcontentratings/weekly/#SportsEvents">Sports Events</a> are ranked by the total number of social TV interactions across Facebook, Instagram Business Accounts and Twitter. Weekly Top Ten lists for <a href="/socialcontentratings/weekly/#Talent">TV Talent</a> are ranked by the total number of social TV engagements across Facebook, Instagram Business Accounts and Twitter.</p>
<p><strong>Interactions</strong><br />
Original social media posts related to a linear TV episode and the engagement with that content.</p>
<p><strong>Engagements</strong><br />
Social media activity generated off of the original posts sent by accounts owned or affiliated with TV programming</p>

$(document).ready(function(){

    $(".ratings-table .sub-bar-arrow").click(function(){
        var table = $(this).closest(".ratings-table");
        $($($(this).closest("tr")).children("td")).children("div.sub-bar").toggleClass("active");
        $(this).closest("tr").find("div.sub-bar-social-icon").toggleClass("active");
        $(this).toggleClass("active");
        console.log('testing123');
        adjustButtons(table);
    });

    $(".table-footer .buttons-footer .expand").click(function() {
        var table = $(this).closest(".ratings-table");
        if (!$(this).is('active')) {
            table.find(".sub-bar").addClass("active");
            table.find('div.sub-bar-social-icon').addClass('active');
            table.find($('.table-footer .buttons-footer span')).removeClass('active');
            $(this).addClass('active');
        }
        adjustButtons(table);
    });

    $(".table-footer .buttons-footer .contract").click(function() {
        var table = $(this).closest(".ratings-table");
        if (!$(this).is('active')) {
            table.find(".sub-bar").removeClass("active");
            table.find('div.sub-bar-social-icon').removeClass('active');
            table.find($('.table-footer .buttons-footer span')).removeClass('active');
            $(this).addClass('active');
        }
        adjustButtons(table);
    });

});

function adjustButtons(ratingsTable) {
    var allBars = ratingsTable.find(".sub-bar").length;
    var activeBars = ratingsTable.find(".sub-bar.active").length;
    
    if ( activeBars == allBars ) {
        ratingsTable.find(".table-footer .buttons-footer .contract").addClass("active");
        ratingsTable.find(".table-footer .buttons-footer .expand").removeClass("active");
    } else if ( activeBars == 0 ) {
        ratingsTable.find(".table-footer .buttons-footer .expand").addClass("active");
        ratingsTable.find(".table-footer .buttons-footer .contract").removeClass("active");
    } else {
        ratingsTable.find(".table-footer .buttons-footer .contract, .table-footer .buttons-footer .expand").addClass("active");
    }
}
